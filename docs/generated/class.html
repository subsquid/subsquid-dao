 class Epoch {
    -BlockNumber offset
    -BlockNumber period
    +new(offset BlockNumber, period BlockNumber) Epoch
    +set_offset(offset BlockNumber)
    +set_period(period BlockNumber)
    +get_offset() BlockNumber
    +get_period_length() BlockNumber
    +get_current_epoch() u32
    +get_current_epoch_since(since BlockNumber) u32
    +get_current_block() u32
 }
 class EpochProxy {
     -Epoch epoch
    +new(epoch Epoch) EpochProxy
    +get() Epoch
 }
 class Registry {
     - Hashmap~Hash~ registry
     - Hashmap~Hash~ commit_name
     - Hashmap~Hash~ commit
     - EpochProxy epoch
     +new(init_value EpochProxy) Registry
     +get_hash(name String) Hash
     +get_indexer_rate() Balance
     +valid(name String) bool
     +available(name String) bool
     +make_commitment(name String, owner AccountId, secret u32) Hash
     +not_expired(hash_name Hash) Option
     +expired(hash_name Hash) Option
     -not_expired_commit(b BlockNumber) bool
     -commit_check(commitment &Hash)
     -commit_register(commitment &Hash)
     +commit(commitment Hash)
     -unlock_balance(name_hash Hash)
     +rent_price(name String, duration u32) Balance
     +get_current_epoch() u32
     +register(name String, from AccountId, duration u32, secret u32)
     -unregister_unchecked(name: Hash)
     +is_owner(name Hash)
     +is_owner_from(name Hash, from AccountId) bool
     +unregister(name: Hash)
 }
class RegistryProxy {
     -Registry registry
    +new(registry Registry) RegistryProxy
    +get() Registry
 }
 class IndexerMeta {

 }

 class Subscription {
    -HashMap~Hash~ subscription
    -RegistryProxy registry
    -EpochProxy epoch
    +new(init_value RegistryProxy, epoch EpochProxy) Subscription
    +get_indexer_rate() Balance
    +get_current_epoch() u32
    -get_subscription_usage(on u32, curr u32) Balance
    -get_min_subscriber_period() BlockNumber

    +is_owner_from(name Hash, from AccountId) bool
    -is_owner(hash_name Hash) bool
    +subscribe(name Hash, from AccountId)
    +unsubscribe(name Hash)
    +get_subscription(name Hash, from AccountId) Option
    +claim_fees(name Hash)
    -claim_fees_unchecked(name Hash)    
 }
 class SubscriptionProxy {
     -Subscription subscription

 class Delegation {
    -HashMap~AccountId~ delegates

    +undelegate(name Hash)

 }
 class DelegationProxy {

 class Dispute {
     -HashMap~Hash~ dispute
     -HashMap~Hash~ reputation
     -Vec~AccountId~ judge
     -HashMap~Hash~ veredict

     +get_current_epoch() -> u32
     +get_reputation(hash_name Hash) -> Option~u16~

     -unchecked_raise_dispute(hash_name Hash, cid Hash, data SubscriberData)
     +withdraw_dispute(name Hash)
     -unchecked_withdraw_dispute(hash_name Hash, cid Hash)

 }
